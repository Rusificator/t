//Задача 5
int main() {
	setlocale(LC_ALL, "Rus");
	const int N = 1000;
	int A[N], B[N], C[N], i, n;
	for (i = 0; i < N; i++)
	{
		A[i] = i; B[i] = 2 * i; C[i] = 0;
	}
	#pragma omp parallel shared(A,B,C) private(i,n)
	{
		n = omp_get_thread_num();
		#pragma omp for
		for (int i = 0; i < N; i++)
		{
			C[i] = A[i] + B[i];
			cout << "Нить" << n << " сложила элементы с номером" << i << endl;
		}
	}
	return 0;
}
